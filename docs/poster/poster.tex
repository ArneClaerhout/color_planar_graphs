		\justifying
		{\Large\bfseries Chromatic numbers}
		\vspace{.3cm}
		
		\begin{definition}
			Chromatic number $\chi_{\mathrm{K}}(\mathcal{P})$ for coloring $\mathrm{K}$, is the minimum needed number of colors to color all planar graphs $G \in \mathcal{P}$ according to $\mathrm{K}$.
		\end{definition}
		
		Chromatic numbers for these coloring variants are not always known.
		\textbf{Conjectures} on the chromatic numbers are tested by exhaustively coloring graphs.
		
		\vspace{1.3cm}
		{\Large\bfseries Algorithm}
		\vspace{.3cm}
		
		Algorithm \ref{alg:search} describes a \textbf{high-level overview} of the designed algorithm.This is a \textbf{backtracking} algorithm using a multitude of optimizations to perform as efficiently as possible while preserving \textbf{expandability}.
		
		\begin{algorithm}[H]
			\caption{\texttt{search\_chromatic\_number}}
			\label{alg:search}
			\textbf{Parameters}: The coloring $K$ and graph $G$
			
			\textbf{Output}: $\chi_{K}(G)$
			\begin{algorithmic}[1]
				\FOR{$max\_c$ {from} $1$ {to} $\mathbf{MAX}$}
				
				\WHILE{not all vertices are colored}
				
				\STATE $v\gets$ \texttt{get\_optimal\_vertex}($G$)
				
				\STATE \texttt{assign\_next\_possible\_color}($v, max\_c$)
				
				\STATE \texttt{update\_neighbors}($G, v$)
				
				\IF{coloring of $G$ is not possible}
					\STATE \texttt{remove\_color}($v$)
				\ENDIF
				
				\ENDWHILE
				
				\ENDFOR
			\end{algorithmic}
		\end{algorithm}
		
		\vspace{0.8cm}
		{\Large\bfseries Results \& Conclusion}
		
		\vspace{0.5cm}
		\begin{block}{Computational evidence for conjectures}
			Conjectures for planar triangulations with number of vertices $n \le 20$ are correct.
		\end{block}
		\begin{itemize}
			\item[\testcheck] First \textbf{open-source} algorithm that colors graphs for these variants.
			\item[\testcheck] Allows for searching of \textbf{gadgets}, such as the one shown in Figure \ref{fig:g_um}, to construct graphs with high chromatic numbers.
			\item[\testcheck] It only performs around $1.4$ times worse than alternatives for the proper coloring. Tools for other colorings do not exist.
		\end{itemize}
