	\begin{frame}
		\frametitle{Conjectures}
		
		\begin{block}{Odd}
			$
			\begin{array}{c@{\hspace{2cm}}c@{\hspace{2cm}}c}
				\begin{aligned}
					& 5 \leq \chi_{\mathrm{O}}(\mathcal{P}) \leq 9 \\
				\end{aligned}
				&
				\pause
				\Large \Rightarrow
				&
				\begin{aligned}
					& \leq \chi_{\mathrm{O}}(\mathcal{P}) = 5 \\
				\end{aligned}
			\end{array}
			$
		\end{block}
		
		\begin{block}{Proper}
			$
			\begin{array}{c@{\hspace{2cm}}c@{\hspace{2cm}}c}
				\begin{aligned}
					& 6 \leq \chi_{\mathrm{PCFo}}(\mathcal{P}) \leq 8 \\
					& \chi_{\mathrm{pCFc}}(\mathcal{P}) = 4 \\
					& 6 \leq \chi_{\mathrm{pUMo}}(\mathcal{P}) \leq 10 \\
					& 6 \leq \chi_{\mathrm{pUMc}}(\mathcal{P}) \leq 8
				\end{aligned}
				&
				\pause
				\Large \Rightarrow
				&
				\begin{aligned}
					&\chi_{\mathrm{PCFo}}(\mathcal{P}) = 6 \\
					&\chi_{\mathrm{pUMo}}(\mathcal{P}) \leq 7 \\
					&\chi_{\mathrm{pUMc}}(\mathcal{P}) = 6
				\end{aligned}
			\end{array}
			$
		\end{block}
	\end{frame}
	
	\begin{frame}
		\frametitle{Conjectures 2}
		
		\begin{block}{Improper}
