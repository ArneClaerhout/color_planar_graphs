		\begin{definition}
			Een graaf $G = (V, E)$ bestaat uit een verzameling $V$ van knopen en een verzameling $E$ van bogen. 
		\end{definition}
		
		\begin{definition}
			Een vlakke graaf is een graaf waarvoor geldt dat de graaf zo getekend kan worden dat geen enkele boog een andere kruist. 
		\end{definition}
		
		\begin{definition}
			De klasse van de vlakke grafen $\mathcal{P}$ is de volledige verzameling van alle grafen die vlak zijn.
		\end{definition}
	\end{frame}
	
	\begin{frame}
		\frametitle{Chromatische getallen}
		
		\begin{definition}
			Het chromatisch getal $\chi(G)$ van een (vlakke) graaf $G$ is het kleinst aantal benodigde kleuren voor het kleuren van de knopen van $G$ zonder twee naburige knopen met eenzelfde kleur.
		\end{definition}
		\begin{itemize}
			\item Andere kleuringsvarianten met andere eisen
			\item Uitbreiden naar $\chi_{K}(G)$
			\item Minimaal aantal benodigde kleuren om $G$ te kleuren voor een kleuring $K$
			
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Chromatische getallen}
		\begin{itemize}
			\begin{definition}
				Het chromatisch getal $\chi_{K}(\mathcal{P})$ voor een kleuring $K$ voor de klasse van vlakke grafen $\mathcal{P}$, is het minimaal benodigd aantal kleuren om een willekeurige graaf binnen die klasse te kunnen kleuren voor de kleuring $K$.
			\end{definition}
			\pause
			\item Bijvoorbeeld: $K_4$ heeft $4$ kleuren bij klassieke kleuring $\implies$ $\chi(\mathcal{P}) \ge 4$. 
			
			Daarna ook bewijs voor volstaan van $4$ kleuren.
		\end{itemize}
		
	\end{frame}
