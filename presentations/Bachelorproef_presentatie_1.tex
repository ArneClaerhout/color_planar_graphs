	\begin{frame}
		\frametitle{Chromatische getallen}
		\begin{itemize}
			\item $\chi(\mathcal{P})$, $\chi_{O}(\mathcal{P})$, $\chi_{pCFo}(\mathcal{P})$, $\dots$ ($\mathcal{P}$ de klasse van vlakke grafen)
			\item Apart per kleuring
			\item Minimaal benodigd aantal kleuren om een willekeurige graaf binnen de klasse van de vlakke grafen te kunnen kleuren.
			
			\item Bijvoorbeeld: $K_4$ heeft $4$ kleuren bij klassieke kleuring $\implies$ $\chi(\mathcal{P}) \ge 4$. 
			
			Daarna ook bewijs voor volstaan van $4$ kleuren.
			
		\end{itemize}
	\end{frame}

	\section{Methodologie}
	
	\begin{frame}
		\frametitle{Methodologie}
		
		\begin{itemize}
			\item Backtracking
			\item Houd mogelijke kleuren bij
		\end{itemize}
		\pause
		\begin{enumerate}
			\item Neem maximumkleur $i = 1$
			\item Kleur een knoop met kleur $\le i$
			\item Check of de kleuring nog klopt
			\item Herhaal 2. en 3. tot alle knopen zijn ingekleurd of de kleuring niet kan
			\item Als de kleuring niet mogelijk is: $i = i + 1$ en ga naar 2.
			\item Als de kleuring wel klopt: Stop.
		\end{enumerate}
